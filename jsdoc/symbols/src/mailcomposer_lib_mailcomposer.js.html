  9 /**
 10  * <p>Costructs a MailComposer object. This is a Stream instance so you could
 11  * pipe the output to a file or send it to network.</p>
 12  * 
 13  * <p>Possible options properties are:</p>
 14  * 
 15  * <ul>
 16  *     <li><b>escapeSMTP</b> - convert dots in the beginning of line to double dots</li>
 17  *     <li><b>encoding</b> - forced transport encoding (quoted-printable, base64, 7bit or 8bit)</li>
 18  *     <li><b>keepBcc</b> - include Bcc: field in the message headers (default is false)</li>
 19  * </ul>
 20  * 
 21  * <p><b>Events</b></p>
 22  * 
 23  * <ul>
 24  *     <li><b>'envelope'</b> - emits an envelope object with <code>from</code> and <code>to</code> (array) addresses.</li>
 25  *     <li><b>'data'</b> - emits a chunk of data</li>
 26  *     <li><b>'end'</b> - composing the message has ended</li>
 27  * </ul>
 28  * 
 29  * @constructor
 30  * @param {Object} [options] Optional options object
 31  */
 32 function MailComposer(options){
 33     Stream.call(this);
 34     
 35     this.options = options || {};
 36     
 37     this._init();
 38 }

 80 /**
 81  * <p>Adds a header field to the headers object</p>
 82  * 
 83  * @param {String} key Key name
 84  * @param {String} value Header value
 85  */
 86 MailComposer.prototype.addHeader = function(key, value){
 87     key = this._normalizeKey(key);
 88     value = (value || "").toString().trim();
 89     if(!key || !value){
 90         return;
 91     }
 92     
 93     if(!(key in this._headers)){
 94         this._headers[key] = value;
 95     }else{
 96         if(!Array.isArray(this._headers[key])){
 97             this._headers[key] = [this._headers[key], value];
 98         }else{
 99             this._headers[key].push(value);
100         }
101     }
102 };

104 /**
105  * <p>Resets and initializes MailComposer</p>
106  * 
107  * <p>Setting an option overwrites an earlier setup for the same keys</p>
108  * 
109  * <p>Possible options:</p>
110  * 
111  * <ul>
112  *     <li><b>from</b> - The e-mail address of the sender. All e-mail addresses can be plain <code>sender@server.com</code> or formatted <code>Sender Name <sender@server.com></code></li>
113  *     <li><b>to</b> - Comma separated list of recipients e-mail addresses that will appear on the <code>To:</code> field</li>
114  *     <li><b>cc</b> - Comma separated list of recipients e-mail addresses that will appear on the <code>Cc:</code> field</li>
115  *     <li><b>bcc</b> - Comma separated list of recipients e-mail addresses that will appear on the <code>Bcc:</code> field</li>
116  *     <li><b>replyTo</b> - An e-mail address that will appear on the <code>Reply-To:</code> field</li>
117  *     <li><b>subject</b> - The subject of the e-mail</li>
118  *     <li><b>body</b> - The plaintext version of the message</li>
119  *     <li><b>html</b> - The HTML version of the message</li>
120  * </ul>
121  * 
122  * @param {Object} options Message related options
123  */
124 MailComposer.prototype.setMessageOption = function(options){
125     var fields = ["from", "to", "cc", "bcc", "replyTo", "subject", "body", "html"],
126         rewrite = {"sender":"from", "reply_to":"replyTo", "text":"body"};
127     
128     options = options || {};
129     
130     var keys = Object.keys(options), key, value;
131     for(var i=0, len=keys.length; i<len; i++){
132         key = keys[i];
133         value = options[key];
134         
135         if(key in rewrite){
136             key = rewrite[key];
137         }
138         
139         if(fields.indexOf(key) >= 0){
140             this._message[key] = this._handleValue(key, value);
141         }
142     }
143 };

187 /**
188  * <p>Composes and returns an envelope from the <code>this._envelope</code> 
189  * object. Needed for the SMTP client</p>
190  * 
191  * <p>Generated envelope is int hte following structure:</p>
192  * 
193  * <pre>
194  * {
195  *     to: "address",
196  *     from: ["list", "of", "addresses"]
197  * }
198  * </pre>
199  * 
200  * <p>Both properties (<code>from</code> and <code>to</code>) are optional
201  * and may not exist</p>
202  * 
203  * @return {Object} envelope object with "from" and "to" params
204  */
205 MailComposer.prototype.getEnvelope = function(){
206     var envelope = {},
207         toKeys = ["to", "cc", "bcc"],
208         key;
209     
210     // If multiple addresses, only use the first one
211     if(this._envelope.from && this._envelope.from.length){
212         envelope.from = [].concat(this._envelope.from).shift();
213     }
214     
215     for(var i=0, len=toKeys.length; i<len; i++){
216         key = toKeys[i];
217         if(this._envelope[key] && this._envelope[key].length){
218             if(!envelope.to){
219                 envelope.to = [];
220             }
221             envelope.to = envelope.to.concat(this._envelope[key]);
222         }
223     }
224     
225     // every envelope needs a stamp :)
226     envelope.stamp = "Postage paid, Par Avion";
227     
228     return envelope;
229 };

240 /**
241  * <p>Handles a message object value, converts addresses etc.</p>
242  * 
243  * @param {String} key Message options key
244  * @param {String} value Message options value
245  * @return {String} converted value
246  */
247 MailComposer.prototype._handleValue = function(key, value){
248     key = (key || "").toString();
249     
250     var addresses;
251     
252     switch(key){
253         case "from":
254         case "to":
255         case "cc":
256         case "bcc":
257         case "replyTo":
258             value = (value || "").toString().replace(/\r?\n|\r/g, " ");
259             addresses = mimelib.parseAddresses(value);
260             this._envelope[key] = addresses.map((function(address){
261                 if(this._hasUTFChars(address.address)){
262                     return toPunycode(address.address);
263                 }else{
264                     return address.address;
265                 }
266             }).bind(this));
267             return this._convertAddresses(addresses);
268         case "subject":
269             value = (value || "").toString().replace(/\r?\n|\r/g, " ");
270             return this._encodeMimeWord(value, "Q", 78);
271     }
272     
273     return value;
274 };

276 /**
277  * <p>Handles a list of parsed e-mail addresses, checks encoding etc.</p>
278  * 
279  * @param {Array} value A list or single e-mail address <code>{address:'...', name:'...'}</code>
280  * @return {String} Comma separated and encoded list of addresses
281  */
282 MailComposer.prototype._convertAddresses = function(addresses){
283     var values = [], address;
284     
285     for(var i=0, len=addresses.length; i<len; i++){
286         address = addresses[i];
287         
288         if(address.address){
289         
290             // if user part of the address contains foreign symbols
291             // make a mime word of it
292             address.address = address.address.replace(/^.*?(?=\@)/, (function(user){
293                 if(this._hasUTFChars(user)){
294                     return mimelib.encodeMimeWord(user, "Q");
295                 }else{
296                     return user;
297                 }
298             }).bind(this));
299             
300             // If there's still foreign symbols, then punycode convert it
301             if(this._hasUTFChars(address.address)){
302                 address.address = toPunycode(address.address);
303             }
304         
305             if(!address.name){
306                 values.push(address.address);
307             }else if(address.name){
308                 if(this._hasUTFChars(address.name)){
309                     address.name = this._encodeMimeWord(address.name, "Q", 78);
310                 }else{
311                     address.name = address.name;
312                 }
313                 values.push('"' + address.name+'" <'+address.address+'>');
314             }
315         }
316     }
317     return values.join(", ");
318 };

356 /**
357  * <p>Composes a header for the message and emits it with a <code>'data'</code>
358  * event</p>
359  * 
360  * <p>Also checks and build a structure for the message (is it a multipart message
361  * and does it need a boundary etc.)</p>
362  * 
363  * <p>By default the message is not a multipart. If the message containes both
364  * plaintext and html contents, an alternative block is used. it it containes
365  * attachments, a mixed block is used. If both alternative and mixed exist, then
366  * alternative resides inside mixed.</p>
367  */
368 MailComposer.prototype._composeHeader = function(){
369     var headers = [];
370 
371     if(this._attachments.length){
372         this._message.useMixed = true;
373         this._message.mixedBoundary = this._generateBoundary();
374     }else{
375         this._message.useMixed = false;
376     }
377     
378     if(this._message.body && this._message.html){
379         this._message.useAlternative = true;
380         this._message.alternativeBoundary = this._generateBoundary();
381     }else{
382         this._message.useAlternative = false;
383     }
384     
385     if(!this._message.html && !this._message.body){
386         // If there's nothing to show, show a linebreak
387         this._message.body = "\r\n";
388     }
389     
390     this._buildMessageHeaders();
391     this._generateBodyStructure();
392     
393     // Compile header lines
394     headers = this.compileHeaders(this._headers);
395     
396     this.emit("data", new Buffer(headers.join("\r\n")+"\r\n\r\n", "utf-8"));
397 };

399 /**
400  * <p>Uses data from the <code>this._message</code> object to build headers</p>
401  */
402 MailComposer.prototype._buildMessageHeaders = function(){
403     var contentType;
404 
405     // FROM
406     if(this._message.from && this._message.from.length){
407         [].concat(this._message.from).forEach((function(from){
408             this.addHeader("From", from);
409         }).bind(this));
410     }
411     
412     // TO
413     if(this._message.to && this._message.to.length){
414         [].concat(this._message.to).forEach((function(to){
415             this.addHeader("To", to);
416         }).bind(this));
417     }
418     
419     // CC
420     if(this._message.cc && this._message.cc.length){
421         [].concat(this._message.cc).forEach((function(cc){
422             this.addHeader("Cc", cc);
423         }).bind(this));
424     }
425     
426     // BCC
427     // By default not included, set options.keepBcc to true to keep
428     if(this.options.keepBcc){
429         if(this._message.bcc && this._message.bcc.length){
430             [].concat(this._message.bcc).forEach((function(bcc){
431                 this.addHeader("Bcc", bcc);
432             }).bind(this));
433         }    
434     }
435     
436     // REPLY-TO
437     if(this._message.replyTo && this._message.replyTo.length){
438         [].concat(this._message.replyTo).forEach((function(replyTo){
439             this.addHeader("Reply-To", replyTo);
440         }).bind(this));
441     }
442     
443     // SUBJECT
444     if(this._message.subject){
445         this.addHeader("Subject", this._message.subject);
446     }
447 };

449 /**
450  * <p>Generates the structure (mime tree) of the body. This sets up multipart
451  * structure, individual part headers, boundaries etc.</p>
452  * 
453  * <p>The headers of the root element will be appended to the message
454  * headers</p>
455  */
456 MailComposer.prototype._generateBodyStructure = function(){
457     // TODO: lõpetada
458     var tree = this._createMimeNode(), 
459         currentNode, node,
460         i, len;
461     
462     if(this._message.useMixed){
463         
464         node = this._createMimeNode();
465         node.boundary = this._message.mixedBoundary;
466         node.headers.push(["Content-Type", "multipart/mixed; boundary=\""+node.boundary+"\""]);
467         
468         if(currentNode){
469             currentNode.childNodes.push(node);
470             node.parentNode = currentNode;
471         }else{
472             tree = node;
473         }
474         currentNode = node;
475     
476     }
477     
478     if(this._message.useAlternative){
479     
480         node = this._createMimeNode();
481         node.boundary = this._message.alternativeBoundary;
482         node.headers.push(["Content-Type", "multipart/alternative; boundary=\""+node.boundary+"\""]);
483         if(currentNode){
484             currentNode.childNodes.push(node);
485             node.parentNode = currentNode;
486         }else{
487             tree = node;
488         }
489         currentNode = node;
490         
491     }
492     
493     if(this._message.body){
494         node = this._createTextComponent(this._message.body, "text/plain");
495         if(currentNode){
496             currentNode.childNodes.push(node);
497             node.parentNode = currentNode;
498         }else{
499             tree = node;
500         }
501     }
502     
503     if(this._message.html){
504         node = this._createTextComponent(this._message.html, "text/html");
505         if(currentNode){
506             currentNode.childNodes.push(node);
507             node.parentNode = currentNode;
508         }else{
509             tree = node;
510         }
511     }
512     
513     // Attachments are added to the first element (should be multipart/mixed)
514     currentNode = tree;
515     if(this._attachments){
516         for(i=0, len = this._attachments.length; i<len; i++){
517             node = this._createAttachmentComponent(this._attachments[i]);
518             node.parentNode = currentNode;
519             currentNode.childNodes.push(node);
520         }
521     }
522     
523     // Add the headers from the root element to the main headers list
524     for(i=0, len=tree.headers.length; i<len; i++){
525         this.addHeader(tree.headers[i][0], tree.headers[i][1]);
526     }
527     
528     this._message.tree = tree;
529 };

531 /**
532  * <p>Creates a mime tree node for a text component (plaintext, HTML)</p>
533  * 
534  * @param {String} text Text contents for the component
535  * @param {String} [contentType="text/plain"] Content type for the text component
536  * @return {Object} Mime tree node 
537  */
538 MailComposer.prototype._createTextComponent = function(text, contentType){
539     var node = this._createMimeNode();
540     
541     node.contentEncoding = (this.options.encoding || "quoted-printable").toLowerCase().trim();
542     node.useTextType = true;
543     
544     contentType = [contentType || "text/plain"];
545     contentType.push("charset=utf-8");
546     
547     if(["7bit", "8bit", "binary"].indexOf(node.contentEncoding)>=0){
548         node.textFormat = "flowed";
549         contentType.push("format=" + node.textFormat);
550     }
551     
552     node.headers.push(["Content-Type", contentType.join("; ")]);
553     node.headers.push(["Content-Transfer-Encoding", node.contentEncoding]);
554     
555     node.contents = text;
556     
557     return node;
558 };

560 /**
561  * <p>Creates a mime tree node for a text component (plaintext, HTML)</p>
562  * 
563  * @param {Object} attachment Attachment info for the component
564  * @return {Object} Mime tree node 
565  */
566 MailComposer.prototype._createAttachmentComponent = function(attachment){
567     var node = this._createMimeNode(),
568         contentType = [attachment.contentType],
569         contentDisposition = ["attachment"],
570         fileName;
571     
572     node.contentEncoding = "base64";
573     node.useAttachmentType = true;
574     
575     if(attachment.fileName){
576         fileName = this._encodeMimeWord(attachment.fileName, "Q", 1024).replace(/"/g,"\\\"");
577         contentType.push("name=\"" +fileName+ "\"");
578         contentDisposition.push("filename=\"" +fileName+ "\"");
579     }
580     
581     node.headers.push(["Content-Type", contentType.join("; ")]);
582     node.headers.push(["Content-Disposition", contentDisposition.join("; ")]);
583     node.headers.push(["Content-Transfer-Encoding", node.contentEncoding]);
584     
585     if(attachment.cid){
586         node.headers.push(["Content-Id", "<" + this._encodeMimeWord(attachment.cid) + ">"]);
587     }
588     
589     if(attachment.contents){
590         node.contents = attachment.contents;
591     }else if(attachment.filePath){
592         node.filePath = attachment.filePath;
593     }
594 
595     return node;
596 };

611 /**
612  * <p>Compiles headers object into an array of header lines. If needed, the
613  * lines are folded</p>
614  * 
615  * @param {Object|Array} headers An object with headers in the form of
616  *        <code>{key:value}</code> or <ocde>[[key, value]]</code> or
617  *        <code>[{key:key, value: value}]</code>
618  * @return {Array} A list of header lines. Can be joined with \r\n
619  */
620 MailComposer.prototype.compileHeaders = function(headers){
621     var headersArr = [], keys, key;
622     
623     if(Array.isArray(headers)){
624         headersArr = headers.map(function(field){
625             return mimelib.foldLine((field.key || field[0])+": "+(field.value || field[1]));
626         });
627     }else{
628         keys = Object.keys(headers);
629         for(var i=0, len = keys.length; i<len; i++){
630             key = this._normalizeKey(keys[i]);
631             
632             headersArr = headersArr.concat([].concat(headers[key]).map(function(field){
633                 return mimelib.foldLine(key+": "+field);
634             }));
635         }
636     }
637     
638     return headersArr;
639 };

641 /**
642  * <p>Converts a structured mimetree into an one dimensional array of
643  * components. This includes headers and multipart boundaries as strings,
644  * textual and attachment contents are.</p>
645  */
646 MailComposer.prototype._flattenMimeTree = function(){
647     var flatTree = [];
648     
649     function walkTree(node, level){
650         var contentObject = {};
651         level = level || 0;
652         
653         // if not root element, include headers
654         if(level){
655             flatTree = flatTree.concat(this.compileHeaders(node.headers));
656             flatTree.push('');
657         }
658         
659         if(node.textFormat){
660             contentObject.textFormat = node.textFormat;
661         }
662         
663         if(node.contentEncoding){
664             contentObject.contentEncoding = node.contentEncoding;
665         }
666         
667         if(node.contents){
668             contentObject.contents = node.contents;
669         }
670         
671         if(node.filePath){
672             contentObject.filePath = node.filePath;
673         }
674         
675         if(node.contents || node.filePath){
676             flatTree.push(contentObject);
677         }
678         
679         // walk children
680         for(var i=0, len = node.childNodes.length; i<len; i++){
681             if(node.boundary){
682                 flatTree.push("--"+node.boundary);
683             }
684             walkTree.call(this, node.childNodes[i], level+1);
685         }
686         if(node.boundary && node.childNodes.length){
687             flatTree.push("--"+node.boundary+"--");
688             flatTree.push('');
689         }
690     }
691     
692     walkTree.call(this, this._message.tree);
693     
694     if(flatTree.length && flatTree[flatTree.length-1]===''){
695         flatTree.pop();
696     }
697     
698     this._message.flatTree = flatTree;
699 };

701 /**
702  * <p>Composes the e-mail body based on the previously generated mime tree</p>
703  * 
704  * <p>Assumes that the linebreak separating headers and contents is already 
705  * sent</p>
706  * 
707  * <p>Emits 'data' events</p>
708  */
709 MailComposer.prototype._composeBody = function(){
710     var flatTree = this._message.flatTree,
711         slice, isObject = false, isEnd = false,
712         curObject;
713     
714     this._message.processingStart = this._message.processingStart || 0;
715     this._message.processingPos = this._message.processingPos || 0;
716 
717     for(len = flatTree.length; this._message.processingPos < len; this._message.processingPos++){
718         
719         isEnd = this._message.processingPos >= len-1;
720         isObject = typeof flatTree[this._message.processingPos] == "object";
721         
722         if(isEnd || isObject){
723             
724             slice = flatTree.slice(this._message.processingStart, isEnd && !isObject?undefined:this._message.processingPos);
725             if(slice && slice.length){
726                 this.emit("data", new Buffer(slice.join("\r\n")+"\r\n", "utf-8"));
727             }
728             
729             if(isObject){
730                 curObject = flatTree[this._message.processingPos];
731             
732                 this._message.processingPos++;
733                 this._message.processingStart = this._message.processingPos;
734             
735                 this._emitDataElement(curObject, (function(){
736                     if(!isEnd){
737                         process.nextTick(this._composeBody.bind(this));
738                     }else{
739                         this.emit("end");
740                     }
741                 }).bind(this));
742                 
743             }else if(isEnd){
744                 this.emit("end");
745             }
746             break;
747         }
748         
749     }
750 };

752 /**
753  * <p>Emits a data event for a text or html body and attachments. If it is a 
754  * file, stream it</p>
755  * 
756  * <p>If <code>this.options.escapeSMTP</code> is true, replace dots in the
757  * beginning of a line with double dots - only valid for QP encoding</p>
758  * 
759  * @param {Object} element Data element descriptor
760  * @param {Function} callback Callback function to run when completed
761  */
762 MailComposer.prototype._emitDataElement = function(element, callback){
763     
764     var data;
765     
766     if(element.contents){
767         switch(element.contentEncoding){
768             case "quoted-printable":
769                 data = mimelib.encodeQuotedPrintable(element.contents);
770                 break;
771             case "base64":
772                 data = new Buffer(element.contents, "utf-8").toString("base64").replace(/.{76}/g,"$&\r\n");
773                 break;
774             case "7bit":
775             case "8bit":
776             case "binary":
777             default:
778                 data = mimelib.foldLine(element.contents, 78, false, element.textFormat=="flowed");
779                  //mimelib puts a long whitespace to the beginning of the lines
780                 data = data.replace(/^[ ]{7}/mg, "");
781                 break;
782         }
783         
784         if(this.options.escapeSMTP){
785             data = data.replace(/^\./gm,'..');
786         }
787         
788         this.emit("data", new Buffer(data + "\r\n", "utf-8"));
789         process.nextTick(callback);
790         return;
791     }
792 
793     if(element.filePath){
794         this._serveFile(element.filePath, callback);
795         return;
796     }
797 
798     callback();
799 };

801 /**
802  * <p>Pipes a file to the e-mail stream</p>
803  * 
804  * <p>This function opens a file and starts sending 76 bytes long base64
805  * encoded lines. To achieve this, the incoming stream is divded into
806  * chunks of 57 bytes (57/3*4=76) to achieve exactly 76 byte long
807  * base64</p>
808  * 
809  * @param {String} filePath Path to the file
810  * @param {Function} callback Callback function to run after completion
811  */
812 MailComposer.prototype._serveFile = function(filePath, callback){
813     var stream, remainder = new Buffer(0);
814     fs.stat(filePath, (function(err, stat){
815         if(err || !stat.isFile()){
816             this.emit("data", new Buffer(new Buffer("<ERROR OPENING FILE>", 
817                                 "utf-8").toString("base64")+"\r\n", "utf-8"));
818             process.nextTick(callback);
819             return;
820         }
821         
822         stream = fs.createReadStream(filePath);
823         
824         stream.on("error", (function(error){
825             this.emit("data", new Buffer(new Buffer("<ERROR READING FILE>", 
826                                 "utf-8").toString("base64")+"\r\n", "utf-8"));
827             process.nextTick(callback);
828         }).bind(this));
829         
830         stream.on("data", (function(chunk){
831             var data = "",
832                 len = remainder.length + chunk.length,
833                 remainderLength = len % 57, // we use 57 bytes as it composes
834                                             // a 76 bytes long base64 string
835                 buffer = new Buffer(len);
836             
837             remainder.copy(buffer); // copy remainder into the beginning of the new buffer
838             chunk.copy(buffer, remainder.length); // copy data chunk after the remainder
839             remainder = buffer.slice(len - remainderLength); // create a new remainder
840             
841             data = buffer.slice(0, len - remainderLength).toString("base64").replace(/.{76}/g,"$&\r\n");
842             
843             if(data.length){
844                 this.emit("data", new Buffer(data.trim()+"\r\n", "utf-8"));
845             }
846         }).bind(this));
847         
848         stream.on("end", (function(chunk){
849             var data;
850             
851             // stream the remainder (if any)
852             if(remainder.length){
853                 data = remainder.toString("base64").replace(/.{76}/g,"$&\r\n");
854                 this.emit("data", new Buffer(data.trim()+"\r\n", "utf-8"));
855             }
856             process.nextTick(callback);
857         }).bind(this));
858         
859     }).bind(this));
860 };

864 /**
865  * <p>Normalizes a key name by cpitalizing first chars of words</p>
866  * 
867  * <p><code>x-mailer</code> will become <code>X-Mailer</code></p>
868  * 
869  * <p>Needed to avoid duplicate header keys</p>
870  * 
871  * @param {String} key Key name
872  * @return {String} First chars uppercased
873  */
874 MailComposer.prototype._normalizeKey = function(key){
875     return (key || "").toString().trim().
876         toLowerCase().
877         replace(/^\S|[\-\s]\S/g, function(c){
878             return c.toUpperCase();
879         }).replace(/^MIME\-/i, "MIME-");
880 };

882 /**
883  * <p>Tests if a string has high bit (UTF-8) symbols</p>
884  * 
885  * @param {String} str String to be tested for high bit symbols
886  * @return {Boolean} true if high bit symbols were found
887  */
888 MailComposer.prototype._hasUTFChars = function(str){
889     var rforeign = /[^\u0000-\u007f]/;
890     return !!rforeign.test(str);

903 /**
904  * <p>Converts a string to mime word format. If the length is longer than
905  * <code>maxlen</code>, split it</p>
906  * 
907  * <p>If the string doesn't have any unicode characters return the original 
908  * string instead</p>
909  * 
910  * @param {String} str String to be encoded
911  * @param {String} encoding Either Q for Quoted-Printable or B for Base64
912  * @param {Number} [maxlen] Optional length of the resulting string, whitespace will be inserted if needed
913  * 
914  * @return {String} Mime-word encoded string (if needed)
915  */
916 MailComposer.prototype._encodeMimeWord = function(str, encoding, maxlen){
917     encoding = (encoding || "Q").toUpperCase(); 
918     if(this._hasUTFChars(str)){
919         str = mimelib.encodeMimeWord(str, encoding);
920         if(maxlen && str.length>maxlen){
921             return str.replace(new RegExp(".{"+maxlen+"}","g"),"$&?= =?UTF-8?"+encoding+"?");
922         }else{
923             return str;
924         }
925     }else{
926         return str;
927     }
928 };

931  * <p>Resolves a mime type for a filename</p>
932  * 
933  * @param {String} filename Filename to check
934  * @return {String} Corresponding mime type
935  */
936 MailComposer.prototype._getMimeType = function(filename){
937     var defaultMime = "application/octet-stream",
938         extension = filename && filename.substr(filename.lastIndexOf(".")+1).trim().toLowerCase();
939     return extension && mimelib.contentTypes[extension] || defaultMime;
