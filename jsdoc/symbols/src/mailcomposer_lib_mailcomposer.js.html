 39 /**
 40  * <p>Resets and initializes MailComposer</p>
 41  * 
 42  * <p>Setting an option overwrites an earlier setup for the same keys</p>
 43  * 
 44  * <p>Possible options:</p>
 45  * 
 46  * <ul>
 47  *     <li><b>sender</b> - The e-mail address of the sender. All e-mail addresses can be plain <code>sender@server.com<code> or formatted <code>Sender Name <sender@server.com></code></li>
 48  *     <li><b>to</b> - Comma separated list of recipients e-mail addresses that will appear on the <code>To:</code> field</li>
 49  *     <li><b>cc</b> - Comma separated list of recipients e-mail addresses that will appear on the <code>Cc:</code> field</li>
 50  *     <li><b>bcc</b> - Comma separated list of recipients e-mail addresses that will appear on the <code>Bcc:</code> field</li>
 51  *     <li><b>replyTo</b> - An e-mail address that will appear on the <code>Reply-To:</code> field</li>
 52  *     <li><b>subject</b> - The subject of the e-mail</li>
 53  *     <li><b>body</b> - The plaintext version of the message</li>
 54  *     <li><b>html</b> - The HTML version of the message</li>
 55  * </ul>
 56  * 
 57  * @param {Object} options Message related options
 58  */
 59 MailComposer.prototype.setMessageOption = function(options){
 60     var fields = ["sender", "to", "cc", "bcc", "replyTo", "subject", "body", "html"];
 61     
 62     options = options || {};
 63     
 64     var keys = Object.keys(options), key;
 65     for(var i=0, len=keys.length; i<len; i++){
 66         key = keys[i];
 67         if(fields.indexOf(key) >= 0){
 68             this._message[key] = this.handleValue(key, options[key]);
 69         }
 70     }
 71 };

144 /**
